// if 조건문
if (불 값이 나오는 표현식) {
  불 값이 참일 때 실행할 문장
}

// if else 조건문
if (불 값이 나오는 표현식) {
  불 값이 참일 때 실행할 문장
} else {
  불 값이 거짓일 때 실행할 문장
}

// if else if 조건문
if (불 표현식) {
  문장
} else if(불 표현식) {
  문장
} else {
  문장
}
